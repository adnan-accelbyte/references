64template <class ObjectType, ESPMode Mode>
65class TSelfPtr
66{
67public:
68 TSelfPtr()
69 {
70 static_assert(!TIsDerivedFrom<ObjectType, TSharedFromThis<ObjectType, ESPMode::ThreadSafe>>::IsDerived, "ObjectType should not derived from TSharedFromThis");
71 static_assert(!TIsDerivedFrom<ObjectType, TSharedFromThis<ObjectType, ESPMode::NotThreadSafe>>::IsDerived, "ObjectType should not derived from TSharedFromThis");

74 virtual ~TSelfPtr()
