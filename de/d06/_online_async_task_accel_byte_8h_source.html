100const static inline FString AsyncTaskCompleteStateToString(const EAccelByteAsyncTaskCompleteState& CompleteState)
101{
102 switch (CompleteState)
103 {
104 case EAccelByteAsyncTaskCompleteState::Success:
105 return TEXT("Success");
106 case EAccelByteAsyncTaskCompleteState::TimedOut:
107 return TEXT("Timed out");
108 case EAccelByteAsyncTaskCompleteState::RequestFailed:
109 return TEXT("Request failed");
110 case EAccelByteAsyncTaskCompleteState::InvalidState:
111 return TEXT("Invalid state");
112 case EAccelByteAsyncTaskCompleteState::Incomplete:
113 return TEXT("Incomplete");
114 }
115 return TEXT("Unknown");

190 // NOTE(Maxwell, 7/8/2021): Due to a bug where we cannot cancel requests on the SDK side, as well as cancel the delegates
191 // that are supposed to run with these requests, if a timeout is quicker than a request could be received from the backend
192 // we may get a crash from that. To combat this for now, we want to set our default timeout to always be one second higher
193 // than the SDK HTTP timeout to give the SDK a chance to fire off its delegates for a timeout.
194 // Fix this once https://accelbyte.atlassian.net/browse/OSS-193 is implemented.
195 TaskTimeoutInSeconds = static_cast<double>(AccelByte::FHttpRetryScheduler::TotalTimeout) + 1.0;

201 virtual void Tick() override
202 {
203 if (HasTaskTimedOut())
204 {
205 UE_LOG(LogAccelByteOSS, Warning, TEXT("Task %s has been idle for longer than %f s"), *GetTaskName(), TaskTimeoutInSeconds);
206 if (bShouldUseTimeout)
207 {
208 CompleteTask(EAccelByteAsyncTaskCompleteState::TimedOut);
209 OnTaskTimedOut();
210 }
211 }
212
213 // If we are not currently in the working state, then kick off the work we need to do for the task
214 if (CurrentState != EAccelByteAsyncTaskState::Working)
215 {
216 CurrentState = EAccelByteAsyncTaskState::Working;
217 OnTaskStartWorking();
218 }
219 }

232 virtual void Initialize() override
233 {
234 CurrentState = EAccelByteAsyncTaskState::Initializing;
235
236 // We only care about setting the last update time if we are using a timeout
237 if (bShouldUseTimeout)
238 {
239 SetLastUpdateTimeToCurrentTime();
240 }
241
242 // Do not attempt to get API clients for server async tasks, as servers do not have API client support.
243 // We also don't need to get the corresponding user ID for the server, as server's don't have user IDs.
244 if (!HasFlag(EAccelByteAsyncTaskFlags::ServerTask))
245 {
246 if (LocalUserNum != INVALID_CONTROLLERID)
247 {
248 GetApiClient(LocalUserNum);
249 }
250 else if (UserId.IsValid())
251 {
252 GetApiClient(UserId.ToSharedRef());
253 }
254
255 // Get corresponding UserId or LocalUserNum for player that started this task
256 GetOtherUserIdentifiers();
257 }
258 }

263 virtual bool HasTaskTimedOut()

367 void CompleteTask(const EAccelByteAsyncTaskCompleteState& InCompleteState)
368 {
369 // We're already marked as complete, do not change this state!
370 if (bIsComplete)
371 {
372 UE_LOG_AB(Warning, TEXT("Tried to complete async task while state was already complete! Current complete state: %s; Requested complete state: %s"), *AsyncTaskCompleteStateToString(CompleteState), *AsyncTaskCompleteStateToString(InCompleteState));
373 return;
374 }
375
376 CurrentState = EAccelByteAsyncTaskState::Completed;
377 CompleteState = InCompleteState;
378 bWasSuccessful = (CompleteState == EAccelByteAsyncTaskCompleteState::Success);
379 bIsComplete = true;

447 void GetOtherUserIdentifiers()
448 {
449 const FOnlineIdentityAccelBytePtr IdentityInterface = StaticCastSharedPtr<FOnlineIdentityAccelByte>(Subsystem->GetIdentityInterface());
450 if (!IdentityInterface.IsValid())
451 {
452 return;
453 }
454
455 // If we have a local user num, then we want to get a user ID from that user num
456 if (LocalUserNum != INVALID_CONTROLLERID)
457 {
458 FUniqueNetIdPtr PlayerId = IdentityInterface->GetUniquePlayerId(LocalUserNum);
459 if (PlayerId.IsValid())
460 {
461#ifdef ONLINESUBSYSTEMACCELBYTE_PACKAGE
462 UserId = FUniqueNetIdAccelByteUser::CastChecked(PlayerId.ToSharedRef());
463#else
464 UserId = StaticCastSharedPtr<const FUniqueNetIdAccelByteUser>(PlayerId);
465#endif
466 }
467 }
468 // Otherwise, if we have a user ID, we want to get their local user num
469 else if (UserId.IsValid())
470 {
471 int32 FoundLocalUserNum;
472 if (IdentityInterface->GetLocalUserNum(UserId.ToSharedRef().Get(), FoundLocalUserNum))
473 {
474 LocalUserNum = FoundLocalUserNum;
475 }
476 }
