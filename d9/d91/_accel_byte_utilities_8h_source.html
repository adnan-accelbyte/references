161 template<typename CharType = TCHAR, template<typename> class PrintPolicy = TPrettyJsonPrintPolicy, typename InStructType>
162 static bool TArrayUStructToJsonString(TArray<InStructType> const& InArray
163 , FString& OutJsonString
164 , int64 CheckFlags = 0
165 , int64 SkipFlags = 0
166 , int32 Indent = 0)
167 {
168 const UStruct* StructDefinition = InStructType::StaticStruct();
169 TArray< TSharedPtr<FJsonValue> > JsonArray;
170
171 for (auto& Item : InArray)
172 {
173 const void* Struct = &Item;
174 TSharedPtr<FJsonObject> JsonObject = MakeShareable(new FJsonObject());
175 if (FJsonObjectConverter::UStructToJsonObject(StructDefinition, Struct, JsonObject.ToSharedRef(), CheckFlags, SkipFlags, nullptr))
176 {
177 TSharedRef< FJsonValueObject > JsonValue = MakeShareable(new FJsonValueObject(JsonObject));
178 JsonArray.Add(JsonValue);
179 }
180 }
181
182 TSharedRef<TJsonWriter<CharType, PrintPolicy<CharType>>> JsonWriter = TJsonWriterFactory<CharType, PrintPolicy<CharType>>::Create(&OutJsonString, Indent);
183 if (FJsonSerializer::Serialize(JsonArray, JsonWriter))
184 {
185 JsonWriter->Close();
186 return true;
187 }
188 else
189 {
190 UE_LOG(LogJson, Warning, TEXT("UStructToFormattedObjectString - Unable to write out json"));
191 JsonWriter->Close();
192 }
193
194 return false;
195 }

286 template<typename ObjectType>
287 static bool UStructArrayToJsonObjectString(TArray<ObjectType> const& Objects
288 , FString& OutString)
289 {
290 OutString.Append(TEXT("["));
291 FString JsonArrayString = TEXT("");
292 for (auto& Item : Objects)
293 {
294 FString Delimiter = JsonArrayString.IsEmpty() ? TEXT("") : TEXT(",");
295 JsonArrayString.Append(Delimiter);
296 FString JsonObjectString = TEXT("");
297 if (!FJsonObjectConverter::UStructToJsonObjectString(Item, JsonObjectString))
298 {
299 return false;
300 }
301 JsonArrayString.Append(JsonObjectString);
302 }
303 OutString.Append(JsonArrayString);
304 OutString.Append(TEXT("]"));
305 return true;
306 }
